def calculate_green_infrastructure_benefits(project_specs):
    """
    Calculate benefits of green infrastructure
    """
    benefits = {}

    # Stormwater reduction
    if 'permeable_pavement' in project_specs:
        benefits['stormwater_reduction'] = '50-90%'

    # Carbon sequestration
    if 'green_roof' in project_specs:
        benefits['carbon_sequestration'] = '5-10 kg/m²/year'

    # Urban heat island mitigation
    if 'urban_forest' in project_specs:
        benefits['temperature_reduction'] = '1-5°C'

    return benefits
